- Tips: use Ajax in some any trick you can do it
        - $.ajax({
            url = "your/url"
            method: "POST" or GET or ETC
            type: "JSON" or ETC
            data: {
                key: value
            }
        }).done(function(data) {
            console.log(data)
        }).fail(function(data) {
            console.log(data)
        }
        - $.post("your/url", { key:value }, function(data) {
            console.log(data)
        })
        - $.get("your/url", { key:value }, function(data) {
            console.log(data)
        })
